\section{Introdução}
\frame{
  \frametitle{Introdução}
  \begin{itemize}
       \vfill
      \item Problema: Falta de jogos que demonstrem o potencial do \textit{input} de uma mesa multi-toque.
      \vfill
      \item Hipótese: Criação de um jogo simulador de física para mesas multi-toque.
      \vfill
      \item Motivação: Um jogo simulador de física consegue utilizar todo potencial do \textit{input} de uma mesa multi-toque,
      sendo perfeito para demonstrar suas capacidades.
   \end{itemize}
}

\frame{
  \frametitle{Introdução}
  \begin{itemize}
   \item Objetivos: Demonstração das funcionalidades de uma mesa multi-toque em um estilo de jogo inovador.
    \vfill
   \item Objetivos específicos: Criação do Deskworld.
    \vfill
   \item Resultados Esperados: Um jogo divertido e inovador que demonstra a capacidade das mesas multi-toque.
\vfill
  \item Metodologia: Criar um jogo em um estilo pouco utilizado mostrando inovações e naturalidade de movimentos.
  \end{itemize}
}

\frame{
  \frametitle{Introdução}
  \begin{itemize}
  \item Este trabalho tem duas partes distintas marcantes:
  \vfill
  	\begin{itemize}
  	\item{Construção de uma mesa multi-toque}
  	\vfill
  	\item{Desenvolvimento do jogo Deskworld}
  	\vfill
		\end{itemize}   
  \end{itemize}
}

   \item São utilizados filtros para captar somete o espectro de luz infravermelha\vfill

   \item Eventos são tratado e interpretados como gestos de acordo com software\vfill
  \end{itemize}
