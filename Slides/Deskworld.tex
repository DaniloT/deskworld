\frame{
	\frametitle{Softwares simuladores de física}
	\begin{itemize}
   \item Um \textit{software} simulador de física consiste de um programa que possui um mundo aberto onde o usuário pode criar o que desejar. Estas criações recebem propriedades de um objeto tradicional e interagem com o mundo de acordo com as leis da física.\vfill
  \end{itemize}
}
