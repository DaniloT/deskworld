\frame{
	\frametitle{\textit{Softwares} simuladores de física}
	\begin{itemize}
   \item \textit{Softwares} simuladores de física permitem criação livre e independente.\vfill
   \item Intuitivos quando em interface sensível ao toque.\vfill
   \item Com algumas modificações, podem ser implementados para número indeterminado de usuários.\vfill
   \item Nenhum dos trabalhos analisados fornece implementação robusta e satisfatória.\vfill
  \end{itemize}
}

	\frametitle{Objetivos}
	\begin{itemize}
   \item Construção de uma mesa com superfície multi-toque.\vfill
   \item Desenvolvimento de um \textit{software} interativo em um gênero pouco explorado focado em um \textit{input} diferente.\vfill
   \item Demonstração de toda a capacidade desse \textit{input} utilizando o software desenvolvido.\vfill
  \end{itemize}
}

\frame{
	\frametitle{Requisitos de interface}
	\begin{itemize}
		\item sua mecânica deve permitir a participação de diversos jogadores simultaneamente;\vfill
		\item deve suportar e processar toques simultâneos;\vfill
		\item deve possuir capacidade de aproximar retas.\vfill
	\end{itemize}
}

\frame{
	\frametitle{Conceito}
	\begin{itemize}
		\item Deixar o usuário criar o mundo que desejar.\vfill
		\item Fornecer ferramentas facilitando sua criação.\vfill
		\item Fornecer maneiras de se modificar o mundo.\vfill
		\item Permitir a criação de vários mundos simultâneos.\vfill
	\end{itemize}
}
